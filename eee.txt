        async function handleBubbleTranslate(msgIdx) {
            const f = customFriends[activeBubbleFriendIndex];
            const msg = f.bubbleMessages[msgIdx];
            
            if (!msg.text || msg.isTranslating) return;

            if (msg.bubbleTranslatedText) {
                msg.bubbleShowTranslation = !msg.bubbleShowTranslation;
                saveFriends();
                renderBubbleMessages();
                return;
            }

            msg.isTranslating = true;
            saveFriends();
            renderBubbleMessages();

            try {
                const configStr = localStorage.getItem('sys_api_config');
                let translated = "";
                if (configStr) {
                    const config = JSON.parse(configStr);
                    if (config.key && config.base) {
                        let fetchUrl = config.base.replace(/\/+$/, '');
                        if (!fetchUrl.endsWith('/chat/completions')) {
                            fetchUrl += (fetchUrl.endsWith('/v1') ? '/chat/completions' : '/v1/chat/completions');
                        }
                        const response = await fetch(fetchUrl, {
                            method: 'POST',
                            headers: {
                                'Content-Type': 'application/json',
                                'Authorization': `Bearer ${config.key}`
                            },
                            body: JSON.stringify({
                                model: config.model || 'gpt-3.5-turbo',
                                messages: [{role: 'user', content: `请将以下内容翻译成精准的中文，只需要返回翻译结果，不要任何多余的解释：\n${msg.text}`}],
                                temperature: 0.3
                            })
                        });
                        if (response.ok) {
                            const data = await response.json();
                            translated = data.choices[0].message.content.trim();
                        } else {
                            translated = "翻译失败：API请求错误 (" + response.status + ")";
                        }
                    } else {
                        translated = "翻译失败：请先配置有效的 API Key";
                    }
                } else {
                    translated = "翻译失败：请先配置 API";
                }
                msg.bubbleTranslatedText = translated;
                msg.bubbleShowTranslation = true; 
                msg.isTranslating = false;
                saveFriends();
                renderBubbleMessages(); 
            } catch (error) {
                msg.bubbleTranslatedText = "翻译请求出错：" + error.message;
                msg.bubbleShowTranslation = true;
                msg.isTranslating = false;
                saveFriends();
                renderBubbleMessages();
            }
        }

        // 爱豆 AI 回复逻辑与随机翻牌 (2-3条渐进式发送)
        async function fetchBubbleAIResponse(friendIndex) {
            const f = customFriends[friendIndex];
            const configStr = localStorage.getItem('sys_api_config');
            const headerNameEl = document.getElementById('bubble-chat-header-name');
            const originalName = f.bubbleName || f.name || "ARTIST";
            
            if (activeBubbleFriendIndex === friendIndex) {
                headerNameEl.innerText = "对方正在输入中...";
            }
            
            if (!configStr) {
                f.bubbleMessages.push({ role: 'ai', type: 'text', text: "[系统提示] 请先配置API", time: getCurrentTimeStr() });
                saveFriends();
                if (activeBubbleFriendIndex === friendIndex) {
                    headerNameEl.innerText = originalName;
                    renderBubbleMessages();
                }
                return;
            }

            const config = JSON.parse(configStr);
            if (!config.key || !config.base) return;

            let messages = [];
            const history = f.bubbleMessages.slice(-20);
            
            let systemPrompt = `你现在是一个爱豆，正在使用类似 Bubble（泡泡）的专属聊天软件与粉丝聊天。
你的角色设定（请仔细读取并严格遵循，这是你聊天的根本）：\n${f.prompt}\n
[核心回复规则]
1. 粉丝非常多，这是像发动态一样的日常聊天，**你必须根据人设把你的回复拆分成3条左右的极简短消息，分开发送**。
2. 绝对不要长篇大论，每条消息只能是非常简短的一句话，符合设定的随性聊天。
3. **韩国人日常聊天习惯：绝对不要使用句号(。)和逗号(，)！遇到需要停顿断句的地方，请一律用空格代替标点符号。**
4. **每条独立的消息之间必须用 "|||" 作为唯一的分隔符！** (例如: 밥 먹었어?|||난 오늘 맛있는 거 먹었는데 ㅎㅎ|||보고싶다 다들)
5. 语气完全贴合人设，不要带任何括号动作描写或机器人的解释说明。直接给出发送的纯文本。`;

            // 查找用户最后一条消息，加入概率翻牌机制
            let lastUserMsg = null;
            for (let i = history.length - 1; i >= 0; i--) {
                if (history[i].role === 'user' && history[i].type === 'text') {
                    lastUserMsg = history[i].text;
                    break;
                }
            }

            // 40% 的几率触发“翻牌”回复
            if (lastUserMsg && Math.random() < 0.4) {
                systemPrompt += `\n5. [触发翻牌] 这次你正好看到了粉丝刚才发的消息：“${lastUserMsg}”。请在你的第一条消息的最开头，使用 “「${lastUserMsg}」\n” （带有全角单引号换行）的格式来引用这句话，然后再接上你的第一句回复！`;
            }

            messages.push({ role: "system", content: systemPrompt });

            // 追加部分历史记忆（仅作语境参考）
            history.forEach(m => {
                let apiRole = m.role === 'ai' ? 'assistant' : 'user';
                messages.push({ role: apiRole, content: m.text || "[图片/语音暂不支持]" });
            });

            try {
                let fetchUrl = config.base.replace(/\/+$/, '');
                if (!fetchUrl.endsWith('/chat/completions')) fetchUrl += (fetchUrl.endsWith('/v1') ? '/chat/completions' : '/v1/chat/completions');

                const response = await fetch(fetchUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${config.key}` },
                    body: JSON.stringify({
                        model: config.model || 'gpt-3.5-turbo',
                        messages: messages,
                        temperature: parseFloat(config.temp) || 1.0,
                    })
                });

                if (response.ok) {
                    const data = await response.json();
                    if (data.choices && data.choices.length > 0) {
                        let aiText = data.choices[0].message.content.trim();
                        
                        // 按照设定好的分隔符分割消息为多条
                        let splitMessages = aiText.split('|||').map(s => s.trim()).filter(s => s.length > 0);
                        
                        // Fallback: 如果 AI 没按照要求加 |||，而是用了普通换行，且行数不超过4行，就按换行拆分
                        if (splitMessages.length === 1 && aiText.includes('\n') && aiText.split('\n').length <= 4) {
                            splitMessages = aiText.split('\n').map(s => s.trim()).filter(s => s.length > 0);
                        }
                        
                        // 异步逐步发出消息 (渐进式发送)
                        const sendProgressively = async () => {
                            for (let i = 0; i < splitMessages.length; i++) {
                                let msgText = splitMessages[i];
                                
                                // 解析 AI 是否成功引用了
                                if (msgText.startsWith('「') && msgText.includes('」')) {
                                    const quoteEndIdx = msgText.indexOf('」');
                                    const quoteText = msgText.substring(1, quoteEndIdx);
                                    const realReply = msgText.substring(quoteEndIdx + 1).trim();
                                    msgText = `<div class="bubble-quote-block">${quoteText}</div>${realReply}`;
                                }
                                
                                f.bubbleMessages.push({ role: 'ai', type: 'text', text: msgText, time: getCurrentTimeStr() });
                                saveFriends();
                                if (activeBubbleFriendIndex === friendIndex) renderBubbleMessages();
                                
                                // 延迟 1~2 秒模拟对方正在输入下一条消息
                                if (i < splitMessages.length - 1) {
                                    if (activeBubbleFriendIndex === friendIndex) headerNameEl.innerText = "对方正在输入中...";
                                    await new Promise(r => setTimeout(r, 1200 + Math.random() * 800));
                                }
                            }
                            if (activeBubbleFriendIndex === friendIndex) headerNameEl.innerText = originalName;
                        };
                        
                        sendProgressively();
                        return; // 提前返回，交由异步循环处理
                    }
                }
            } catch (err) {
                f.bubbleMessages.push({ role: 'ai', type: 'text', text: `[网络错误] ${err.message}`, time: getCurrentTimeStr() });
                saveFriends();
                if (activeBubbleFriendIndex === friendIndex) renderBubbleMessages();
            }
            if (activeBubbleFriendIndex === friendIndex) headerNameEl.innerText = originalName;
        }

        async function pullModels() {
            const base = document.getElementById('api-base').value.trim();
            const key = document.getElementById('api-key').value.trim();
            const btn = document.getElementById('btn-pull-models');
            
            if (!base || !key) { alert("请先填写 API 地址和 API Key"); return; }
            
            let fetchUrl = base.replace(/\/+$/, '');
            if (!fetchUrl.endsWith('/models')) {
                fetchUrl += (fetchUrl.endsWith('/v1') ? '/models' : '/v1/models');
            }
            
            btn.innerText = "正在拉取...";
            try {
                const response = await fetch(fetchUrl, {
                    method: 'GET',
                    headers: { 'Authorization': `Bearer ${key}` }
                });
                
                if (!response.ok) {
                    const errText = await response.text();
                    throw new Error(`HTTP 错误: ${response.status} - ${errText.substring(0, 50)}`);
                }
                
                const data = await response.json();
                const modelSelect = document.getElementById('api-model');
                
                modelSelect.innerHTML = '';
                
                if (data.data && Array.isArray(data.data)) {
                    data.data.forEach(m => {
                        const opt = document.createElement('option');
                        opt.value = m.id;
                        opt.innerText = m.id;
                        modelSelect.appendChild(opt);
                    });
                    alert(`成功拉取 ${data.data.length} 个模型!`);
                    saveSysSettings();
                } else {
                    alert("获取的数据格式不符合 OpenAI 标准:\n" + JSON.stringify(data).substring(0, 100));
                }
            } catch (err) {
                alert("拉取模型失败 (请检查地址、网络或跨域设置): \n" + err.message);
            } finally {
                btn.innerText = "拉取模型";
            }
        }

        function getVoiceDuration(text) {
            const chars = text ? text.length : 1;
            let secs = Math.ceil(chars * 0.25); 
            if (secs < 1) secs = 1;
            if (secs > 599) secs = 599; 
            return Math.floor(secs / 60) + ":" + String(secs % 60).padStart(2, '0');
        }

        let playingVoiceIdx = null;
        let voiceInterval = null;

        function playVoiceSafe(e, idx) {
            if (isMultiSelectMode) return;
            if (isLongPressTriggered) return;

            const bg = document.getElementById(`voice-bg-${idx}`);
            const btn = document.getElementById(`voice-btn-${idx}`);
            if (!bg || !btn) return;

            if (playingVoiceIdx === idx) {
                clearInterval(voiceInterval);
                bg.style.width = '0%';
                btn.innerHTML = '<path d="M8 5v14l11-7z"/>';
                playingVoiceIdx = null;
                return;
            }

            if (playingVoiceIdx !== null) {
                const oldBg = document.getElementById(`voice-bg-${playingVoiceIdx}`);
                const oldBtn = document.getElementById(`voice-btn-${playingVoiceIdx}`);
                if (oldBg) oldBg.style.width = '0%';
                if (oldBtn) oldBtn.innerHTML = '<path d="M8 5v14l11-7z"/>';
                clearInterval(voiceInterval);
            }

            const f = customFriends[activeFriendIndex];
            const msg = f.messages[idx];
            const durStr = msg.duration || getVoiceDuration(msg.text);
            const parts = durStr.split(':');
            const totalSecs = parseInt(parts[0]) * 60 + parseInt(parts[1]);

            playingVoiceIdx = idx;
            btn.innerHTML = '<rect x="6" y="5" width="4" height="14"/><rect x="14" y="5" width="4" height="14"/>';
            
            let progress = 0;
            bg.style.transition = 'none';
            bg.style.width = '0%';
            
            const tickRate = 50; 
            const increment = 100 / (totalSecs * (1000 / tickRate));

            setTimeout(() => {
                bg.style.transition = 'width 0.05s linear';
                voiceInterval = setInterval(() => {
                    progress += increment;
                    if (progress >= 100) {
                        progress = 100;
                        bg.style.width = progress + '%';
                        clearInterval(voiceInterval);
                        setTimeout(() => {
                            bg.style.transition = 'width 0.2s ease';
                            bg.style.width = '0%';
                            btn.innerHTML = '<path d="M8 5v14l11-7z"/>';
                            playingVoiceIdx = null;
                        }, 300);
                    } else {
                        bg.style.width = progress + '%';
                    }
                }, tickRate);
            }, 50);
        }

        function sendMessage() {
            const input = document.getElementById('chat-input');
            const text = input.value.trim();
            if(!text) {
                if (activeFriendIndex !== -1) fetchAIResponse(activeFriendIndex);
                return;
            }
            
            const f = customFriends[activeFriendIndex];
            const timeStr = getCurrentTimeStr();
            
            f.messages.push({ role: 'user', type: 'text', text: text, time: timeStr });
            input.value = "";
            toggleSendIcon(); 
            document.getElementById('chat-attach-menu').style.display = 'none';
            saveFriends();
            renderMessages();
            updateChatTime(activeFriendIndex);
            
            setTimeout(() => {
                const container = document.getElementById('chat-content');
                if (container) container.scrollTop = container.scrollHeight;
            }, 50);
        }

        async function triggerAutoMessage(friendIndex) {
            const f = customFriends[friendIndex];
            const configStr = localStorage.getItem('sys_api_config');
            if (!configStr) return;
            const config = JSON.parse(configStr);
            if (!config.key || !config.base) return;

            let messages = [];
            const now = new Date();
            const dateOptions = { year: 'numeric', month: 'long', day: 'numeric', weekday: 'long' };
            const timeOptions = { hour: '2-digit', minute: '2-digit' };
            const currentDateStr = now.toLocaleDateString('zh-CN', dateOptions);
            const currentTimeStr = now.toLocaleTimeString('zh-CN', timeOptions);

            let systemPrompt = `[系统参考] 当前时间：${currentDateStr} ${currentTimeStr}。\n[特殊主动指令] 你大概已经有超过3个小时没和对方说话了。请你主动开启新的话题、表达关心或分享日常。
[聊天习惯要求]
1. 必须将回复拆分成3条左右的极简短消息发出，绝对不要长篇大论，每条只能有一句话。
2. 韩国人日常聊天习惯：绝对不要使用句号(。)和逗号(，)，一律用空格代替断句。
3. 消息之间必须用 "|||" 分隔！语气完全贴合角色设定，不可带有任何机器动作或解释。\n`;
            
            const globalPersona = localStorage.getItem('kkt_my_persona');
            if (globalPersona) systemPrompt += `我的全局人设（对方）：\n${globalPersona}\n`;

            if (f.prompt) systemPrompt += `你的角色设定：\n${f.prompt}\n`;
            if (f.persona) systemPrompt += `我针对此对话的特定人设补充：\n${f.persona}\n`;
            if (f.worldbook) systemPrompt += `世界观：\n${f.worldbook}\n`;

            if (systemPrompt.trim()) {
                messages.push({ role: "system", content: systemPrompt });
            }

            const memoryLimit = parseInt(config.memory) || 50;
            const history = f.messages.slice(-memoryLimit);
            history.forEach(m => {
                let apiRole = m.role === 'ai' ? 'assistant' : 'user';
                
                if (m.type === 'image' && m.imgUrl) {
                    messages.push({
                        role: apiRole,
                        content: [
                            { type: "text", text: "请识别并结合人设回复这张图片：" },
                            { type: "image_url", image_url: { url: m.imgUrl } }
                        ]
                    });
                } else if (m.type === 'photo_mock') {
                    messages.push({ role: apiRole, content: `[用户发送了一张照片，内容描述为：${m.text}]` });
                } else if (m.type === 'voice') {
                    messages.push({ role: apiRole, content: `[用户发送一条语音，内容为：${m.text}]` });
                } else if (m.type === 'transfer') {
                    messages.push({ role: apiRole, content: `[收到一笔转账：${m.amount} 韩元，备注：${m.note || '无'}]` });
                } else {
                    messages.push({ role: apiRole, content: m.text });
                }
            });

            try {
                let fetchUrl = config.base.replace(/\/+$/, '');
                if (!fetchUrl.endsWith('/chat/completions')) {
                    fetchUrl += (fetchUrl.endsWith('/v1') ? '/chat/completions' : '/v1/chat/completions');
                }

                const response = await fetch(fetchUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${config.key}` },
                    body: JSON.stringify({
                        model: config.model || 'gpt-3.5-turbo',
                        messages: messages,
                        temperature: parseFloat(config.temp) || 1.0,
                        top_p: 1.0,
                        frequency_penalty: 0,
                        presence_penalty: 0
                    })
                });

                if (response.ok) {
                    const data = await response.json();
                    if (data.choices && data.choices.length > 0) {
                        const aiText = data.choices[0].message.content.trim();
                        f.messages.push({ role: 'ai', type: 'text', text: aiText, time: getCurrentTimeStr() });
                    } else {
                        f.messages.push({ role: 'ai', type: 'text', text: `[API返回异常: 未找到回复内容]`, time: getCurrentTimeStr() });
                    }
                } else {
                    const errData = await response.text();
                    let errMsg = `[API错误: ${response.status}]`;
                    try {
                        const parsedErr = JSON.parse(errData);
                        if (parsedErr.error && parsedErr.error.message) errMsg += ` ${parsedErr.error.message}`;
                    } catch(e) { errMsg += ` ${errData.substring(0, 60)}...`; }
                    f.messages.push({ role: 'ai', type: 'text', text: errMsg, time: getCurrentTimeStr() });
                }
            } catch (err) {
                f.messages.push({ role: 'ai', type: 'text', text: `[网络请求失败: ${err.message}] 请检查API地址或跨域设置。`, time: getCurrentTimeStr() });
            }

            updateChatTime(friendIndex);
            saveFriends();
            if (activeFriendIndex === friendIndex) {
                headerNameEl.innerText = originalName;
                renderMessages();
            }
        }

        async function translateVoiceMsg(idx) {
            const f = customFriends[activeFriendIndex];
            const msg = f.messages[idx];
            if (msg.translatedText) return;

            try {
                const configStr = localStorage.getItem('sys_api_config');
                let translated = "翻译失败：请配置API";
                if (configStr) {
                    const config = JSON.parse(configStr);
                    if (config.key && config.base) {
                        let fetchUrl = config.base.replace(/\/+$/, '');
                        if (!fetchUrl.endsWith('/chat/completions')) {
                            fetchUrl += (fetchUrl.endsWith('/v1') ? '/chat/completions' : '/v1/chat/completions');
                        }
                        const response = await fetch(fetchUrl, {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${config.key}` },
                            body: JSON.stringify({
                                model: config.model || 'gpt-3.5-turbo',
                                messages: [{role: 'user', content: `请将以下内容翻译成精准的中文，只需要返回翻译结果，不要任何多余的解释：\n${msg.text}`}],
                                temperature: 0.3
                            })
                        });
                        if (response.ok) {
                            const data = await response.json();
                            translated = data.choices[0].message.content.trim();
                        } else {
                            translated = "翻译失败：API请求错误";
                        }

                msg.showTranslation = true;
                saveFriends();
                const transDiv = document.getElementById(`voice-trans-${idx}`);
                if(transDiv) transDiv.innerText = translated;
                else renderMessages();
            } catch (e) {
                msg.translatedText = "翻译出错";
                msg.showTranslation = true;
                saveFriends();
                renderMessages();
            }
        }

        async function handleTranslate(element, msgIdx) {

            const msg = f.messages[msgIdx];

            if (msg.translatedText) {
                msg.showTranslation = !msg.showTranslation;
                saveFriends();
                renderMessages();
                return;

            const originalText = msg.text;
            element.innerHTML = `<div class="original-text" style="line-height: 1.5;">${originalText}</div><div class="translate-divider"></div><div class="translated-text">翻译中...</div>`;

            try {
                const configStr = localStorage.getItem('sys_api_config');
                let translated = "";
                if (configStr) {
                    const config = JSON.parse(configStr);
                    if (config.key && config.base) {
                        let fetchUrl = config.base.replace(/\/+$/, '');
                        if (!fetchUrl.endsWith('/chat/completions')) {
                            fetchUrl += (fetchUrl.endsWith('/v1') ? '/chat/completions' : '/v1/chat/completions');
                        }
                        const response = await fetch(fetchUrl, {
                            method: 'POST',
                            headers: {
                                'Content-Type': 'application/json',
                                'Authorization': `Bearer ${config.key}`
                            },
                            body: JSON.stringify({
                                model: config.model || 'gpt-3.5-turbo',
                                messages: [{role: 'user', content: `请将以下内容翻译成精准的中文，只需要返回翻译结果，不要任何多余的解释：\n${originalText}`}],
                                temperature: 0.3
                            })
                        });
                        if (response.ok) {
                            const data = await response.json();
                            translated = data.choices[0].message.content.trim();
                        } else {
                            translated = "翻译失败：API请求错误 (" + response.status + ")";
                        }

                        translated = "翻译失败：请先配置有效的 API Key";
                    }
                } else {
                    translated = "翻译失败：请先配置 API";
                }
                msg.translatedText = translated;
                msg.showTranslation = true; 
                saveFriends();
                renderMessages(); 
            } catch (error) {
                msg.translatedText = "翻译请求出错：" + error.message;
                msg.showTranslation = true;
                saveFriends();
                renderMessages();
            }
        }

        function handleFile(e) {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = (ev) => {
                const url = ev.target.result;
                if (themeTarget === 'banner-img') {
                    document.getElementById('banner-img').style.backgroundImage = `url('${url}')`;
                    localStorage.setItem('img_banner', url);
                } else if (themeTarget === 'avatar-img') {
                    document.getElementById('avatar-img').style.backgroundImage = `url('${url}')`;
                    document.getElementById('k-my-ava').style.backgroundImage = `url('${url}')`;
                    document.getElementById('k-detail-ava').style.backgroundImage = `url('${url}')`;
                    localStorage.setItem('img_avatar-img', url);
                    
                    if (document.getElementById('set-my-ava')) document.getElementById('set-my-ava').style.backgroundImage = `url('${url}')`;
                    renderMessages();
                } else if (themeTarget === 'bubble-my-ava') {
                    document.getElementById('bubble-my-ava').style.backgroundImage = `url('${url}')`;
                    localStorage.setItem('bubble_my_ava', url);
                } else if (themeTarget === 'wallpaper') {
                    document.body.style.backgroundImage = `url('${url}')`;
                    localStorage.setItem('img_wallpaper', url);
                    alert("桌面壁纸已更新");
                } else if (themeTarget === 'chat-wallpaper') {
                    document.documentElement.style.setProperty('--chat-bg-img', `url('${url}')`);
                    localStorage.setItem('img_chat_wallpaper', url);
                    alert("聊天壁纸已更新");
                } else if (themeTarget === 'new-friend-ava') {
                    tempNewFriendAva = url;
                    document.getElementById('new-friend-ava').style.backgroundImage = `url('${url}')`;
                } else if (themeTarget === 'set-friend-ava') {
                    document.getElementById('set-friend-ava').style.backgroundImage = `url('${url}')`;
                    if (activeFriendIndex !== -1) {
                        customFriends[activeFriendIndex].ava = url;
                        saveFriends();
                        renderCustomFriends();
                        renderMessages();
                    } else if (activeBubbleFriendIndex !== -1) {
                        customFriends[activeBubbleFriendIndex].ava = url;
                        saveFriends();
                        renderBubbleFriends();
                        renderBubbleMessages();
                    }
                } else if (themeTarget === 'k-detail-bg') {
                    document.getElementById('k-detail-bg').style.backgroundImage = `url('${url}')`;
                    localStorage.setItem('k_detail_bg', url);
                } else if (themeTarget === 'k-detail-ava') {
                    document.getElementById('k-detail-ava').style.backgroundImage = `url('${url}')`;
                    document.getElementById('avatar-img').style.backgroundImage = `url('${url}')`;
                    document.getElementById('k-my-ava').style.backgroundImage = `url('${url}')`;
                    localStorage.setItem('img_avatar-img', url);
                } else if (themeTarget === 'new-post-img') {
                    tempNewPostImg = url;
                    document.getElementById('new-post-img-preview').style.backgroundImage = `url('${url}')`;
                    document.getElementById('new-post-img-preview').innerHTML = '';
                } else if (themeTarget && themeTarget.startsWith('icon-')) {
                    document.getElementById(themeTarget).style.backgroundImage = `url('${url}')`;
                    document.getElementById(themeTarget).innerHTML = '';
                    
                    const prevId = 'prev-' + themeTarget;
                    if(document.getElementById(prevId)) {
                        document.getElementById(prevId).style.backgroundImage = `url('${url}')`;
                        document.getElementById(prevId).innerHTML = '';
                    }
                    localStorage.setItem('img_' + themeTarget, url);
                }
            };
            reader.readAsDataURL(file);
            e.target.value = '';
        }

        window.onload = function() {
            // Load theme images
            const ids = ['banner-img', 'avatar-img', 'wallpaper', 'chat-wallpaper', 'icon-settings', 'icon-theme', 'icon-kakao', 'icon-ins', 'icon-bubble', 'icon-weibo', 'icon-x', 'icon-weverse', 'k_detail_bg', 'bubble_my_ava'];
            ids.forEach(id => {
                const stored = localStorage.getItem(id.startsWith('img_') || id.startsWith('k_') || id.startsWith('bubble_') ? id : 'img_' + id);
                if (stored) {
                    if (id === 'avatar-img') {
                        document.getElementById('avatar-img').style.backgroundImage = `url('${stored}')`;
                        document.getElementById('k-my-ava').style.backgroundImage = `url('${stored}')`;
                        if(document.getElementById('k-detail-ava')) document.getElementById('k-detail-ava').style.backgroundImage = `url('${stored}')`;
                    } else if (id === 'wallpaper') {
                        document.body.style.backgroundImage = `url('${stored}')`;
                    } else if (id === 'chat-wallpaper') {
                        document.documentElement.style.setProperty('--chat-bg-img', `url('${stored}')`);
                    } else if (id === 'k_detail_bg') {
                        if(document.getElementById('k-detail-bg')) document.getElementById('k-detail-bg').style.backgroundImage = `url('${stored}')`;
                    } else if (id === 'bubble_my_ava') {
                        if(document.getElementById('bubble-my-ava')) document.getElementById('bubble-my-ava').style.backgroundImage = `url('${stored}')`;
                    } else if (id.startsWith('icon-')) {
                        const el = document.getElementById(id);
                        if(el) { el.style.backgroundImage = `url('${stored}')`; el.innerHTML = ''; }
                        const prevEl = document.getElementById('prev-' + id);
                        if(prevEl) { prevEl.style.backgroundImage = `url('${stored}')`; prevEl.innerHTML = ''; }
                    } else {
                        const el = document.getElementById(id);
                        if (el) el.style.backgroundImage = `url('${stored}')`;
                    }
                }
            });

            // Load texts
            const homeName = localStorage.getItem('home_nickname');
            if (homeName) { document.getElementById('txt-nickname').innerText = homeName; }
            const kakaoName = localStorage.getItem('kkt_my_name');
            if (kakaoName) { 
                document.getElementById('k-my-name').innerText = kakaoName; 
                if(document.getElementById('k-detail-name')) document.getElementById('k-detail-name').innerText = kakaoName;

            if (bStatus) document.getElementById('bubble-my-status').innerText = bStatus;

            loadSysSettings();
            applyFontSettings();
            renderCustomFriends();
            renderBubbleFriends();
            renderKakaoPosts();

            // Status Bar clock
            setInterval(() => {
                const now = new Date();
                document.getElementById('status-time').innerText = now.getHours().toString().padStart(2, '0') + ':' + now.getMinutes().toString().padStart(2, '0');
            }, 1000);
            
            document.addEventListener('click', (e) => {
                const addMenu = document.getElementById('kakao-add-menu');
                if (addMenu && addMenu.style.display === 'block' && !e.target.closest('.icons')) {
                    addMenu.style.display = 'none';
                }
            });
        };
